**setup.sh** (Bash, macOS/Linux):
```bash
#!/usr/bin/env bash
set -euo pipefail

echo "ComfyUI Template Agent - Setup"
echo "=============================="

# Check Python 3.12+
python_version=$(python3 --version 2>/dev/null | grep -oP '\d+\.\d+' | head -1)
if [[ -z "$python_version" ]]; then
    echo "ERROR: Python 3 not found. Install Python 3.12+"
    exit 1
fi
major=$(echo "$python_version" | cut -d. -f1)
minor=$(echo "$python_version" | cut -d. -f2)
if (( major < 3 || (major == 3 && minor < 12) )); then
    echo "ERROR: Python $python_version found, need 3.12+."
    exit 1
fi
echo "OK Python $python_version"

# Install dependencies
echo "Installing dependencies..."
pip install -e ".[dev]"

# Create skills symlinks
SKILLS_DIR="$HOME/.claude/skills"
REPO_SKILLS="$(cd "$(dirname "$0")" && pwd)/.claude/skills"

mkdir -p "$SKILLS_DIR"

for skill in comfy-discover comfy-templates comfy-validate comfy-compose comfy-document comfy-flow; do
    target="$SKILLS_DIR/$skill"
    source="$REPO_SKILLS/$skill"
    if [[ -L "$target" ]]; then
        echo "SKIP $skill (symlink exists)"
    elif [[ -d "$target" ]]; then
        echo "WARN $skill directory exists at $target — skipping (remove manually to re-link)"
    else
        ln -s "$source" "$target"
        echo "OK   $skill -> $source"
    fi
done

# Check for ComfyUI MCP server
echo ""
echo "Prerequisites:"
echo "  - ComfyUI MCP server (comfyui-cloud or comfyui-mcp) for compose skill"
echo "  - GITHUB_TOKEN env var for higher API rate limits (optional)"
echo ""

# Run tests
echo "Running tests..."
python -m pytest tests/ -q
echo ""
echo "Setup complete!"
```
